function changeBackgroundColor() {
  
  var myParagraph = document.getElementById("p1");

  var myText = myParagraph.textContent;

  if (myText === " C") {
    myParagraph.style.backgroundColor = "green";
  } else if (myText === "C") {
    myParagraph.style.backgroundColor = "red";
  }
