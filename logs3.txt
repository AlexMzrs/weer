$(document).ready(function () {
  $(".navContainer").load("nav.html");
  $(".route").load("pages/Reverification.html", function () {
    // Attach the event listener after the content is loaded
    $("#data-select").change(onSelectChanged);
  });

  $(document).on("click", ".button", function () {
    const text = $(this).html();
    navigator.clipboard.writeText(text);
  });

  // Initialize the content div visibility
  $(".content-div").hide();

  // New event listener for the Vnav-button clicks
  $(document).on("click", ".Vnav-button", function () {
    const contentId = $(this).text().replace(" ", "_");

    // Hide all content divs
    $(".content-div").hide();
    $("#" + contentId).show();

    // Toggle active state for buttons
    $(".Vnav-button").removeClass("active");
    $(this).addClass("active");
  });
});

function loadPage(button) {
  text = button.innerHTML;
  $(".route").load("pages/" + text + ".html", function () {
    // Reattach the event listener after loading a new page
    $("#data-select").change(onSelectChanged);
  });
}

function requestDetails(value) {

  $("#req3").html(
    "Kindly check your secured inbox for all the necessary instructions regarding your " +
      value +
      " change request.\nYou can take your time to gather and submit the required documents whenever it's most convenient for you.\nUpon receiving the documents, we'll move forward with your requested."
  );
}
function onSelectChanged() {
  const selectedValue = $("#data-select").val();
  let selectedText = $("#data-select option:selected").text();
  $(".steps > div").addClass("hidden");
  $(".steps > div." + selectedValue).removeClass("hidden");

  if (typeof requestDetails === "function") {
    requestDetails(selectedText);
  }
}
function cdcPost(type) {
  let selectedText = $("#data-select option:selected").text();
  let post;
  if (type == 1)
    post =
      "Request: " +
      selectedText +
      " Change\nAction: Asked for documents via SI / Informed cst about the process. \nOutcome: Waiting for cst documents";
  else if (type == 2)
    post =
      "Request: " +
      selectedText +
      " Change\nAction: Cst sent documents/ BO-Ticket raised / informed cst via SI. \nOutcome: Waiting for ticket completion";
  else if (type == 3)
    post =
      "Request: " +
      selectedText +
      " Change\nAction: BO-Ticket completed /" +
      selectedText +
      " successfully changed/ informed cst via SI. \nOutcome: Cst assisted / case closed / nfa";
  else if (type == 4)
    post =
      "Request: " +
      selectedText +
      " Change\nAction: successfully performed verification and " +
      selectedText +
      " change / informed cst in chat. \nOutcome: Cst assisted / case closed / nfa";
  navigator.clipboard.writeText(post);
}
function downgradeSteps() {
  $("#panel div").addClass("hidden");
  const country = $("#dgselect").val();
  const membership = $("#dgselect-memb").val();
  const signUpDate = new Date($("#sdate").val());

  const europeDate = new Date("2021,02,16");
  const italyDate = new Date("2021,02,25");
  const spainDate = italyDate;
  const franceDate = new Date("2021,03,11");
  const germanyDate = new Date("2021,03,18");
  const austriaDate = germanyDate;

  const header = $("#dgheader");

  if (
    signUpDate < europeDate ||
    (country == "it" && signUpDate < italyDate) ||
    (country == "sp" && signUpDate < spainDate) ||
    (country == "fr" && signUpDate < franceDate) ||
    (country == "ger" && signUpDate < germanyDate) ||
    (country == "at" && signUpDate < austriaDate)
  ) {
    header.html("Old T&C").css("color", "red");
  } else header.html("New T&C").css("color", "green");

  if (header.text() == "New T&C") {
    if (membership == "You") $("#panel #youNew").removeClass("hidden");
    else if (membership == "Smart") $("#panel #smartNew").removeClass("hidden");
    else if (membership == "Metal") $("#panel #metalNew").removeClass("hidden");
  } else if (header.text() == "Old T&C") {
    if (membership == "You") $("#panel #youOld").removeClass("hidden");
    else if (membership == "Smart") $("#panel #smartOld").removeClass("hidden");
    else if (membership == "Metal") $("#panel #metalOld").removeClass("hidden");
  }
}
