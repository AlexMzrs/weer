function GetEmail(type)
{
    const cstName = document.getElementById("inputLA").value;
    switch(type)
    {

        case 'LaReq': LARequest(cstName);break;
        case 'LaEsc': LAEscalated(cstName);break;
        case 'LaComp': LACompleted(cstName);break;
        case 'LaRej' : LARejected(cstName);break;

        case 'NatReq' : NaRequest(cstName);break;
        case 'NatEsc' : NaEscalated(cstName);break;
        case 'NatComp': NaCompleted(cstName);break;
        case 'NatRej' : NaRejected(cstName);break;


        case 'DocReq' : DoRequest(cstName);break;
        case 'DocEsc' : DoEscalated(cstName);break;
        case 'DocComp': DoCompleted(cstName);break;
        case 'DocRej' : DoRejected(cstName);break;

        case 'GenReq' : GenRequest(cstName);break;
        case 'GenEsc' : GenEscalated(cstName);break;
        case 'GenComp': GenCompleted(cstName);break;
        case 'GenRej' : GenRejected(cstName);break;

        case 'LnReq' : LnRequest(cstName);break;
        case 'LnEsc' : LnEscalated(cstName);break;
        case 'LnComp': LnCompleted(cstName);break;
        case 'LnRej' : LnRejected(cstName);break;

        case 'TaReq' : TaRequest(cstName);break;
        case 'TaEsc' : TaEscalated(cstName);break;
        case 'TaComp': TaCompleted(cstName);break;
        case 'TaRej' : TaRejected(cstName);break;
       
       
       
    } 
}

function getName(name) {
   
  const nameParts = name.split(' ').map(part => {
    var fullName = part.charAt(0).toUpperCase() + part.slice(1).toLowerCase();
    return fullName;
  });
  return nameParts.join(' ');
}
